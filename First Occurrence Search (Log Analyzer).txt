#include <stdio.h>
int main() {
    char str[100], ch;
    int index = -1;
    printf("Enter string: ");
    scanf("%s", str);
    printf("Enter character to search: ");
    scanf(" %c", &ch);   
    for(int i = 0; str[i] != '\0'; i++) {
        if(str[i] == ch) {
            index = i;
            break;      
        }
    }
    printf("%d\n", index);
    return 0;
}
