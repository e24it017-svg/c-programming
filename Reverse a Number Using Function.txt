#include <stdio.h>
int reverseNumber(int n) {
    int reversed = 0;
    while (n != 0) {
        int digit = n % 10;          
        reversed = reversed * 10 + digit; 
        n = n / 10;                 
    }
    return reversed;
}
int main() {
    int n;
    scanf("%d", &n);
    int result = reverseNumber(n);
    printf("%d", result);
    return 0;
}
