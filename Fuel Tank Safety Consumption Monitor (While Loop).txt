#include <stdio.h>
int main() {
    int a, b, c;        
    int i = 0;
    int count = 0;      
    scanf("%d", &a);    
    scanf("%d", &b);    
    while(i < b) {
        scanf("%d", &c);   
        if(a >= c) {       
            a -= c;        
            count++;       
        } else {
            break;         
        }
        i++;
    }
    printf("Completed Trips: %d\n", count);
    printf("Remaining Fuel: %d\n", a);
    return 0;
}
