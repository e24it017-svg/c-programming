#include<stdio.h>
int main(){
    int n,i;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    int mlen=1;
    int clen=1;
    for(i=1;i<n;i++){
        if(a[i]>a[i-1]){
        clen++;
        }else{
        clen=1;
        }
        if(clen<mlen){
        mlen=clen; 
        }
    }
    printf("%d",mlen);
    return 0;
}
