#include <stdio.h>
int main() {
    int a, b;          
    int i = 0;
    int inactive = 0;  
    scanf("%d", &a);   
    while(i < a) {
        scanf("%d", &b);   
        if(b == 0)       
            inactive++;
        i++;
    }
    printf("Inactive Weeks: %d\n", inactive);

    if(inactive >= 3)
        printf("Risk Status: High\n");
    else
        printf("Risk Status: Low\n");
    return 0;
}
