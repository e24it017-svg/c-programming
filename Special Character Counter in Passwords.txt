#include <stdio.h>
#include <ctype.h>
int main() {
    char password[100];
    int count = 0;
    printf("Enter Password: ");
    scanf("%s", password);
    for(int i = 0; password[i] != '\0'; i++) {
        if(!isalnum(password[i])) {   
            count++;
        }
    }
    printf("Number of special characters: %d\n", count);
    return 0;
}
