#include <stdio.h>
int main() {
    int a, b;        
    int i = 0;
    int streak = 0;    
    int max = 0;       
    scanf("%d", &a);   
    while(i < a) {
        scanf("%d", &b);   
        if(b == 0) {       
            streak++;
            if(streak > max)
                max = streak;
        } else {           
            streak = 0;   
        }

        i++;
    }
    printf("Longest Default Streak: %d\n", max);
    return 0;
}
