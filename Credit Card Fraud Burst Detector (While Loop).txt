#include <stdio.h>
int main() {
    int N, a[100], b = 0, c = 0, i = 0, d = -1;
    scanf("%d", &N);
    while (i < N) {
        scanf("%d", &a[i]);
        i++;
    }
    i = 0;
    while (i < N) {
        if (a[i] >= 50000) {
            b++;
            c++;
            if (c == 3 && d == -1) d = i + 1;
        } else c = 0;
        i++;
    }
    if (d != -1) printf("Fraud Triggered At Attempt: %d\n", d);
    else printf("Fraud Triggered At Attempt: Not Triggered\n");
    printf("High-Value Transactions: %d\n", b);
    return 0;
}
