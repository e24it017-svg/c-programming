#include<stdio.h>
int main(){
    int i,n,a,Total=0,count=0;
    printf("input:");
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a);
        Total=Total+a;
        if(a>100){
        count++;
        }
   }
   printf("Total Loss:%d\n",Total); 
   printf("High Loss Days:%d",count);
    return 0;
}
