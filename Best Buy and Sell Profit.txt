#include <stdio.h>
int main()
{
    int n, i;
    int price[100];
    int min, profit, maxProfit;
    scanf("%d", &n);
    for(i = 0; i < n; i++)
       scanf("%d", &price[i]);
    min = price[0];
    maxProfit = price[1] - price[0];
    for(i = 1; i < n; i++)
    {
        profit = price[i] - min;
        if(profit > maxProfit)
            maxProfit = profit;
        if(price[i] < min)
            min = price[i];
    }
    printf("%d", maxProfit);
    return 0;
}
