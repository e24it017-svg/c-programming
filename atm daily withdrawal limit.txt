#include <stdio.h>
int main() {
    int n,a,total=0;
    printf("input:");
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a);
        total=total+a;
    }
    if(total>10000){
        printf("Limit Exceeded");
    }
    else{
        printf("Approved");
    }

    return 0;
}
