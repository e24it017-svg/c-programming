#include <stdio.h>
int main() {
    int a, b, c;     
    int d;           
    int e = 0;       
    scanf("%d", &a);
    scanf("%d", &b);
    d = a;
    for(int i = 0; i < b; i++) {
        scanf("%d", &c);
        d = d - c;
        if(d < 5000) {
            e++;
        }
    }
    printf("Remaining Cash: %d\n", d);
    printf("Risk Count: %d\n", e);
    return 0;
}
