#include <stdio.h>
int main() {
    int a, b, c;    
    int total = 0;
    int overflow_days = 0;
    scanf("%d", &a);
    scanf("%d", &b);
    for(int i = 0; i < b; i++) {
        scanf("%d", &c);
        if(total <= a && total + c > a) {
            overflow_days++;
        }
        total += c;
    }
    printf("Remaining Capacity: %d\n", a - total);
    printf("Overflow Days: %d\n", overflow_days);
    return 0;
}
