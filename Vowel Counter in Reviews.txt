#include <stdio.h>
#include <string.h>
int main() {
    char review[100];
    int i, count = 0;
    char ch;
    printf("Enter review: ");
    scanf("%s", review);
    for(i = 0; review[i] != '\0'; i++) {
        ch = review[i];
        if(ch >= 'A' && ch <= 'Z') {
            ch = ch + ('a' - 'A');
        }
        if(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
            count++;
        }
    }
    printf("%d\n", count);
    return 0;
}
