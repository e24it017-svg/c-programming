#include <stdio.h>
#include <string.h>
int main() {
    char code[100];
    int length, i, isPalindrome = 1;
    printf("Enter product code: ");
    scanf("%s", code);
    length = strlen(code);
    for(i = 0; i < length / 2; i++) {
        if(code[i] != code[length - 1 - i]) {
            isPalindrome = 0;
            break;
        }
    }
    if(isPalindrome)
        printf("Palindrome\n");
    else
        printf("Not Palindrome\n");
    return 0;
}
