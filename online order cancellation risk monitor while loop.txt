#include <stdio.h>
int main() {
    int n, a, b = 0, c = 0, i = 0; 
    scanf("%d", &n);
    while(i < n) {
        scanf("%d", &a);      
        if(a == 1)
            b++;              
        else
            c++;              
        i++;
    }
    printf("Successful: %d\n", b);
    printf("Cancelled: %d\n", c);
    if(c > b)
        printf("Status: Risk\n");
    else
        printf("Status: Safe\n");
    return 0;
}
