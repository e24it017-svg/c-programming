#include <stdio.h>
int main() {
    int a, b = 1, c, d, e = 0, f = 0, g = 0;
    scanf("%d", &a);
    scanf("%d", &c);
    d = c;
    while (b < a) {
        scanf("%d", &c);
        if (c < d) {
            e++;
            f++;
            if (f == 3 && g == 0) {
                g = b + 1;
            }
        } else {
            f = 0;
        }
        d = c;
        b++;
    }
    if (g == 0) {
        printf("Crash Day: Not Detected\n");
    } else {
        printf("Crash Day: %d\n", g);
    }
    printf("Total Drops: %d\n", e);
    return 0;
}
