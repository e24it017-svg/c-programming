#include <stdio.h>
int main() {
    int a, b;          
    int c[100];        
    int d = 0;         
    int e = 0;        
    scanf("%d", &a);
    scanf("%d", &b);
    for(int i = 0; i < b; i++) {
        scanf("%d", &c[i]);
    }
    while(d < b) {
        a = a - c[d];
        if(a <= 0) {
            if(a < 0)
                e = -a;   
            else
                e = 0;
            printf("Exhausted Day: %d\n", d + 1);
            printf("Overused Data: %d\n", e);
            return 0;
        }
        d = d + 1;
    }
    printf("Exhausted Day: Not Exhausted\n");
    printf("Overused Data: 0\n");
    return 0;
}
