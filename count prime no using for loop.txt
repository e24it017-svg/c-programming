#include <stdio.h>
int main() {
    int i,j,n,count=0,prime;
    printf("input:");
    scanf("%d",&n);
    for(i=2;i<=n;i++){
        prime =1;
        for(j=2;j<=i/2;j++){
            if(i%j==0){
                prime=0;
                break;
            }
        }
        if(prime==1){
            count++;
        }
    }
    printf("%d",count);
    return 0;
}
