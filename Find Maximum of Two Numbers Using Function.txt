
#include<stdio.h>
void max(int a,int b){
    if(a>b){
        printf("first number is larger");
    }
    else if(b>a){
        printf("%d is greater than %d ",b,a);
    }
    else{
        printf("both number are equal");
    }
        
}
int main(){
    int x,y;
    scanf("%d%d",&x,&y);
    max(x,y);
    return 0;
}
