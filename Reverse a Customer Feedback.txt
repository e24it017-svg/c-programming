#include <stdio.h>
#include <string.h>
int main() {
    char feedback[100];
    int length, i;
    char reversed[100];
    printf("Enter feedback: ");
    scanf("%s", feedback);
    length = strlen(feedback);
    for(i = 0; i < length; i++) {
        reversed[i] = feedback[length - 1 - i];
    }
    reversed[length] = '\0'; 
    printf("%s\n", reversed);
    return 0;
}
