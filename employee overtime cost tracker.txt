#include <stdio.h>
int main() {
    int a;        
    int b;        
    int c = 0;   
    int d = 0;    
    scanf("%d", &a);
    for(int i = 0; i < a; i++) {
        scanf("%d", &b);
        c = c + b;
        if(b > 3) {
            d++;
        }
    }
    printf("Total Overtime Hours: %d\n", c);
    printf("Overtime Cost: %d\n", c * 200);
    printf("Heavy Overtime Days: %d\n", d);
    return 0;
}
