#include<stdio.h>
int main()
{
    int n,i;
    int a[n];
    scanf("%d",&n);
    for(i=0;i<n;i++){
    scanf("%d",&a[i]);
    }
    int flag=1;
    for(i=0;i<n/2;i++){
        if(a[i]!=a[n-i-1]){
            flag=0;
            break;
        }
    }
    if(flag==1)
    printf("Yes");
    else
    printf("No");
    return 0;
    
}
