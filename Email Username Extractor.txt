#include <stdio.h>
#include <string.h>
int main() {
    char email[100];
    char username[100];
    printf("Enter email: ");
    scanf("%s", email);
    int i = 0;
    while(email[i] != '@' && email[i] != '\0') {
        username[i] = email[i];
        i++;
    }
    username[i] = '\0'; 
    printf("%s\n", username);
    return 0;
}
