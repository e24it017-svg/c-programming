#include <stdio.h>
#include <string.h>
int main() {
    char text[200];
    int i, count = 0;
    char ch;
    printf("Enter transcript: ");
    scanf("%s", text); 
    for(i = 0; text[i] != '\0'; i++) {
        ch = text[i];
        if(ch >= 'A' && ch <= 'Z') {
            ch = ch + ('a' - 'A');
        }
        if((ch >= 'a' && ch <= 'z') && !(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u')) {
            count++;
        }
    }
    printf("%d\n", count);
    return 0;
}
