#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int prices[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &prices[i]);
    }
    int maxFromRight = -1;  
    for(int i = n - 1; i >= 0; i--) {
        int current = prices[i];
        prices[i] = maxFromRight;   
        if(current > maxFromRight)
            maxFromRight = current; 
    }
    for(int i = 0; i < n; i++) {
        printf("%d ", prices[i]);
    }
    return 0;
}
