#include <stdio.h>
int main() {
    int N;
    scanf("%d", &N); 
    int transactions[N];
    for (int i = 0; i < N; i++) {
        scanf("%d", &transactions[i]); 
    }
    int firstRepeat = -1; 
    for (int i = 0; i < N; i++) {
        for (int j = i + 1; j < N; j++) {
            if (transactions[i] == transactions[j]) {
                firstRepeat = transactions[i]; 
                break; 
            }
        }
        if (firstRepeat != -1) {
            break; 
        }
    }
    printf("%d\n", firstRepeat); 
    return 0;
}
