#include <stdio.h>
int isPalindrome(int n) {
    int original = n;
    int reversed = 0;
    while (n != 0) {
        int digit = n % 10;
        reversed = reversed * 10 + digit;
        n = n / 10;
    }
    if (original == reversed)
        return 1;
    else
        return 0;
}
int main() {
    int n;
    scanf("%d", &n);
    if (isPalindrome(n))
        printf("Palindrome");
    else
        printf("Not Palindrome");
    return 0;
}
