#include<stdio.h>
int main(){
    int i,n,a,Total=0,count=0;
    printf("input:");
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a);
        Total=Total+a;
        if(a<40){
        count++;
        }
   }
   int avg=Total/n;
   printf("Avg Score:%d\n",avg); 
   printf("Failed Subjects:%d",count);
    return 0;
}
