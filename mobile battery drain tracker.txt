#include <stdio.h>
int main() {
    int b, n;
    printf("input:");
    scanf("%d", &b);
    scanf("%d", &n);
    int d, total = 0;
    for(int i = 0; i < n; i++) {
        scanf("%d", &d);
        total += d;
    }
    int r = b - total;
    if(r < 0) {
        r= 0;
    }
    printf("Remaining Battery: %d%%", r);
    return 0;
}
