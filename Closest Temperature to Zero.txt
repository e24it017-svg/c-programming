#include <stdio.h>
#include <stdlib.h>
int main()
{
    int n, i;
    int temp[100];
    int closest;
    scanf("%d", &n);
    for(i = 0; i < n; i++)
    {
        scanf("%d", &temp[i]);
    }
    closest = temp[0];
    for(i = 1; i < n; i++)
    {
        if(abs(temp[i]) < abs(closest))
        {
            closest = temp[i];
        }
        else if(abs(temp[i]) == abs(closest) && temp[i] > closest)
        {
            closest = temp[i];
        }
    }
    printf("%d", closest);
    return 0;
}
