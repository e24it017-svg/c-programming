#include <stdio.h>
int main() {
    int a, b = 0, c = 0, d = 0, e = 0;
    scanf("%d", &a);
    while (b < a) {
        scanf("%d", &d);
        c = c + d;
        if (d > 4) {
            e++;
        }
        b++;
    }
    printf("Total Overtime: %d\n", c);
    printf("Burnout Days: %d\n", e);
    return 0;
}
